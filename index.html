const SS = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { return handleAction(e.parameter); } function handleAction(data) { const action = data.action; if (action === 'getPedidos') { const sheet = SS.getSheetByName('Pedidos'); const rows = sheet.getDataRange().getValues(); if (rows.length < 2) return json({ok:true, pedidos:[]}); const h = rows[0]; return json({ok:true, pedidos: rows.slice(1).map(r => obj(h,r))}); } if (action === 'nuevoPedido') { const sheet = SS.getSheetByName('Pedidos'); if (sheet.getLastRow() === 0) sheet.appendRow(['ID','Fecha','Hora','Area','Solicitante','Proveedor','Items','Obs','Estado','Monto','FechaEntrega']); sheet.appendRow([ data.id, data.fecha, data.hora, data.area, data.solicitante, data.proveedor, data.items, data.obs, 'pendiente', 0, '' ]); return json({ok:true, msg:'Pedido guardado'}); } if (action === 'actualizarEstado') { const sheet = SS.getSheetByName('Pedidos'); const values = sheet.getDataRange().getValues(); for (let i = 1; i < values.length; i++) { if (String(values[i][0]) === String(data.id)) { sheet.getRange(i+1, 9).setValue(data.estado); if (data.monto) sheet.getRange(i+1, 10).setValue(data.monto); if (data.fechaEntrega) sheet.getRange(i+1, 11).setValue(data.fechaEntrega); return json({ok:true, msg:'Actualizado'}); } } return json({ok:false, error:'Pedido no encontrado'}); } if (action === 'getPagos') { const sheet = SS.getSheetByName('Pagos'); const rows = sheet.getDataRange().getValues(); if (rows.length < 2) return json({ok:true, pagos:[]}); const h = rows[0]; return json({ok:true, pagos: rows.slice(1).map(r => obj(h,r))}); } if (action === 'nuevoPago') { const sheet = SS.getSheetByName('Pagos'); if (sheet.getLastRow() === 0) sheet.appendRow(['ID','Fecha','Proveedor','Monto','Medio','RegistradoPor','Obs']); sheet.appendRow([ data.id, data.fecha, data.proveedor, data.monto, data.medio, data.registradoPor, data.obs ]); return json({ok:true, msg:'Pago registrado'}); } return json({ok:false, error:'Accion desconocida'}); } function obj(headers, row) { const o = {}; headers.forEach((h, i) => o[h] = row[i]); return o; } function json(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
